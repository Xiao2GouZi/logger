```js
const str = "Hello!";
for (const s of str) {
  console.log(s);
}
```

```js
"use strict";
var str = "Hello!";
for (var _i = 0, str_1 = str; _i < str_1.length; _i++) {
    var s = str_1[_i];
    console.log(s);
}
```
